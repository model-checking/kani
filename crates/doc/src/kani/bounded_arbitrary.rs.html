4//! This module introduces implementations for some std containers.
5
6use kani::{Arbitrary, BoundedArbitrary};
7
8impl<T: Arbitrary> BoundedArbitrary for Box<[T]> {
9    fn bounded_any<const N: usize>() -> Self {
10        let len: usize = kani::any_where(|l| *l <= N);
11        // The following is equivalent to:
12        // ```
13        // (0..len).map(|_| T::any()).collect()
14        // ```
15        // but leads to more efficient verification
16        let mut b = Box::<[T]>::new_uninit_slice(len);
17        for i in 0..len {
18            b[i] = std::mem::MaybeUninit::new(T::any());
19        }
20        unsafe { b.assume_init() }
21    }
22}
23
24// This implementation overlaps with `kani::any_vec` in `kani/library/kani/src/vec.rs`.
25// This issue `https://github.com/model-checking/kani/issues/4027` tracks deprecating
26// `kani::any_vec` in favor of this implementation.
27impl<T: Arbitrary> BoundedArbitrary for Vec<T> {
28    fn bounded_any<const N: usize>() -> Self {
29        let real_length = kani::any_where(|&size| size <= N);
30        let array: [T; N] = kani::any();
31        let mut vec = Vec::from(array);
32        vec.truncate(real_length);
33        vec
34    }
35}

37impl BoundedArbitrary for String {
38    fn bounded_any<const N: usize>() -> Self {
39        let bytes: [u8; N] = kani::any();
40
41        if let Some(s) = bytes.utf8_chunks().next() { s.valid().into() } else { String::new() }
42    }
43}

45impl<K, V> BoundedArbitrary
46    for std::collections::HashMap<K, V, std::hash::BuildHasherDefault<std::hash::DefaultHasher>>
47where
48    K: Arbitrary + std::cmp::Eq + std::hash::Hash,
49    V: Arbitrary,
50{
51    fn bounded_any<const N: usize>() -> Self {
52        let mut hash_map = std::collections::HashMap::default();
53        for _ in 0..N {
54            // this check seems to perform better than 0..kany::any_where(|l| *l <= N)
55            if bool::any() {
56                hash_map.insert(K::any(), V::any());
57            }
58        }
59        hash_map
60    }
61}

63impl<V> BoundedArbitrary

65where

68    fn bounded_any<const N: usize>() -> Self {
69        let mut hash_set = std::collections::HashSet::default();
70        for _ in 0..N {
71            // this check seems to perform better than 0..kany::any_where(|l| *l <= N)
72            if bool::any() {
73                hash_set.insert(V::any());
74            }
75        }
76        hash_set
