4//! This macro generates implementations of the `KaniIntoIter` trait for various common types that are used in `for` loops.
5//! We use this trait to overwrite the Rust IntoIter trait to reduce call stacks and avoid complicated loop invariant specifications,
6//! while maintaining the semantics of the loop.
7use crate::{KaniIntoIter, KaniPtrIter};
8
9impl<T: Clone> KaniIntoIter for Vec<T> {
10    type Iter = KaniPtrIter<T>;
11    fn kani_into_iter(self) -> Self::Iter {
12        let s = self.iter();
13        KaniPtrIter::new(s.as_slice().as_ptr(), s.len())
14    }
15}
