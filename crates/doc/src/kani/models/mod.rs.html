3//! Contains definitions that Kani compiler may use to model functions that are not suitable for
4//! verification or functions without a body, such as intrinsics.
5//!
6//! Note that these are models that Kani uses by default; thus, we keep them separate from stubs.
7
8// Definitions in this module are not meant to be visible to the end user, only the compiler.
9#[allow(dead_code)]

14    /// Similar definition to portable SIMD.
15    /// We cannot reuse theirs since TRUE and FALSE defs are private.
16    /// We leave this private today, since this is not necessarily a final solution, so we don't
17    /// want users relying on this.
18    /// Our definitions are also a bit more permissive to comply with the platform intrinsics.
19    pub(super) trait MaskElement: PartialEq + Debug {
20        const TRUE: Self;
21        const FALSE: Self;
22    }
23
24    macro_rules! impl_element {
25        { $ty:ty } => {
26            impl MaskElement for $ty {
27                const TRUE: Self = -1;
28                const FALSE: Self = 0;
29            }
30        }
31    }
32
33    macro_rules! impl_unsigned_element {
34        { $ty:ty } => {
35            impl MaskElement for $ty {
36                // Note that in the declaration of the intrinsic it is documented that the lane
37                // values should be -1 or 0:
38                // <https://github.com/rust-lang/rust/blob/338cfd3/library/portable-simd/crates/core_simd/src/intrinsics.rs#L134-L144>
39                //
40                // However, MIRI and the Rust compiler seems to accept unsigned values and they
41                // use their binary representation. Thus, that's what we use for now.
42                /// All bits are 1 which represents TRUE.
43                const TRUE: Self = <$ty>::MAX;
44                /// All bits are 0 which represents FALSE.
45                const FALSE: Self = 0;
46            }
47        }
48    }

64    /// Calculate the minimum number of lanes to represent a mask
65    /// Logic similar to `bitmask_len` from `portable_simd`.
66    /// <https://github.com/rust-lang/portable-simd/blob/490b5cf/crates/core_simd/src/masks/to_bitmask.rs#L75-L79>
67    pub(super) const fn mask_len(len: usize) -> usize {
68        len.div_ceil(8)
69    }
70
71    #[cfg(target_endian = "little")]
72    unsafe fn simd_bitmask_impl<T, const LANES: usize>(input: &[T; LANES]) -> [u8; mask_len(LANES)]
73    where
74        T: MaskElement,
75    {
76        let mut mask_array = [0; mask_len(LANES)];
77
78        // The implementation below is the equivalent of the following:
79        // ```rust
80        //     for lane in (0..input.len()).rev() {
81        //         let byte = lane / 8;
82        //         let mask = &mut mask_array[byte];
83        //         let shift_mask = *mask << 1;
84        //         *mask = if input[lane] == T::TRUE {
85        //             shift_mask | 0x1
86        //         } else {
87        //             assert_eq!(input[lane], T::FALSE, "Masks values should either be 0 or -1");
88        //             shift_mask
89        //         };
90        //     }
91        // ```
92        // but is intentionally written in a way that minimizes the number of
93        // loop iterations. In particular, it's implemented as a nested loop
94        // where the outer loop iterates over bytes and the inner "loop" (which
95        // is manually unwound) iterates over bits in a byte.  This is to avoid
96        // needing a high unwind value for harnesses that invoke this code (e.g.
97        // through the `HashSet` data structure).
98        for (byte_idx, byte) in mask_array.iter_mut().enumerate() {
99            // Calculate the starting lane for this byte
100            let start_lane = byte_idx << 3;
101            // Calculate how many bits to process (handle the last byte which might be partial)
102            let bits_to_process = (LANES - start_lane).min(8);
103
104            *byte = if bits_to_process > 0 && input[start_lane] == T::TRUE { 1 << 0 } else { 0 }
105                | if bits_to_process > 1 && input[start_lane + 1] == T::TRUE { 1 << 1 } else { 0 }
106                | if bits_to_process > 2 && input[start_lane + 2] == T::TRUE { 1 << 2 } else { 0 }
107                | if bits_to_process > 3 && input[start_lane + 3] == T::TRUE { 1 << 3 } else { 0 }
108                | if bits_to_process > 4 && input[start_lane + 4] == T::TRUE { 1 << 4 } else { 0 }
109                | if bits_to_process > 5 && input[start_lane + 5] == T::TRUE { 1 << 5 } else { 0 }
110                | if bits_to_process > 6 && input[start_lane + 6] == T::TRUE { 1 << 6 } else { 0 }
111                | if bits_to_process > 7 && input[start_lane + 7] == T::TRUE { 1 << 7 } else { 0 };
112
113            assert!(
114                bits_to_process < 1
115                    || input[start_lane] == T::TRUE
116                    || input[start_lane] == T::FALSE,
117                "Masks values should either be 0 or -1"
118            );
119            assert!(
120                bits_to_process < 2
121                    || input[start_lane + 1] == T::TRUE
122                    || input[start_lane + 1] == T::FALSE,
123                "Masks values should either be 0 or -1"
124            );
125            assert!(
126                bits_to_process < 3
127                    || input[start_lane + 2] == T::TRUE
128                    || input[start_lane + 2] == T::FALSE,
129                "Masks values should either be 0 or -1"
130            );
131            assert!(
132                bits_to_process < 4
133                    || input[start_lane + 3] == T::TRUE
134                    || input[start_lane + 3] == T::FALSE,
135                "Masks values should either be 0 or -1"
136            );
137            assert!(
138                bits_to_process < 5
139                    || input[start_lane + 4] == T::TRUE
140                    || input[start_lane + 4] == T::FALSE,
141                "Masks values should either be 0 or -1"
142            );
143            assert!(
144                bits_to_process < 6
145                    || input[start_lane + 5] == T::TRUE
146                    || input[start_lane + 5] == T::FALSE,
147                "Masks values should either be 0 or -1"
148            );
149            assert!(
150                bits_to_process < 7
151                    || input[start_lane + 6] == T::TRUE
152                    || input[start_lane + 6] == T::FALSE,
153                "Masks values should either be 0 or -1"
154            );
155            assert!(
156                bits_to_process < 8
157                    || input[start_lane + 7] == T::TRUE
158                    || input[start_lane + 7] == T::FALSE,
159                "Masks values should either be 0 or -1"
160            );
161        }
162
163        mask_array
164    }

166    /// Stub for simd_bitmask.
167    ///
168    /// It will reduce a simd vector (TxN), into an integer of size S (in bits), where S >= N.
169    /// Each bit of the output will represent a lane from the input. A lane value of all 0's will be
170    /// translated to 1b0, while all 1's will be translated to 1b1.
171    ///
172    /// In order to be able to do this pragmatically, we take additional parameters that are filled
173    /// by the compiler.
174    #[rustc_diagnostic_item = "KaniModelSimdBitmask"]
175    pub(super) unsafe fn simd_bitmask<T, U, E, const LANES: usize>(input: T) -> U
176    where
177        [u8; mask_len(LANES)]: Sized,
178        E: MaskElement,
179    {
180        // These checks are compiler sanity checks to ensure we are not doing anything invalid.
181        assert_eq!(
182            size_of::<U>(),
183            size_of::<[u8; mask_len(LANES)]>(),
184            "Expected size of return type and mask lanes to match",
185        );
186        assert_eq!(
187            size_of::<T>(),
188            size_of::<Simd::<E, LANES>>(),
189            "Expected size of input and lanes to match",
190        );
191
192        unsafe {
193            let data = &*(&input as *const T as *const [E; LANES]);
194            let mask = simd_bitmask_impl(data);
195            (&mask as *const [u8; mask_len(LANES)] as *const U).read()
196        }
197    }
198
199    /// Structure used for sanity check our parameters.
200    #[repr(simd)]
201    struct Simd<T, const LANES: usize>([T; LANES]);
202}

210    /// Test that the `simd_bitmask` model is equivalent to the intrinsic for all true and all false
211    /// masks with lanes represented using i16.
212    #[test]
213    fn test_bitmask_i16() {
214        check_portable_bitmask::<_, i16, 16, u16>(mask16x16::splat(false));
215        check_portable_bitmask::<_, i16, 16, u16>(mask16x16::splat(true));
216    }
217
218    /// Tests that the model correctly fails if an invalid value is given.
219    #[test]
220    #[should_panic(expected = "Masks values should either be 0 or -1")]
221    fn test_invalid_bitmask() {
222        let invalid_mask = unsafe { mask32x16::from_int_unchecked(i32x16::splat(10)) };
223        assert_eq!(
224            unsafe { kani_intrinsic::simd_bitmask::<_, u16, i32, 16>(invalid_mask) },
225            u16::MAX
226        );
227    }
228
229    /// Tests that the model correctly fails if the size parameter of the mask doesn't match the
230    /// expected number of bytes in the representation.
231    #[test]
232    #[should_panic(expected = "Expected size of return type and mask lanes to match")]
233    fn test_invalid_generics() {
234        let mask = mask32x16::splat(false);
235        assert_eq!(unsafe { kani_intrinsic::simd_bitmask::<_, u16, i32, 2>(mask) }, u16::MAX);
236    }

238    /// Test that the `simd_bitmask` model is equivalent to the intrinsic for a few random values.
239    /// These values shouldn't be symmetric and ensure that we also handle endianness correctly.
240    #[test]
241    fn test_bitmask_i32() {
242        check_portable_bitmask::<_, i32, 8, u8>(mask32x8::from([
243            true, true, false, true, false, false, false, true,
244        ]));
245
246        check_portable_bitmask::<_, i32, 4, u8>(mask32x4::from([true, false, false, true]));

261    /// Compare the value returned by our model and the portable simd representation.
262    fn check_portable_bitmask<T, E, const LANES: usize, M>(mask: Mask<T, LANES>)
263    where
264        T: std::simd::MaskElement,
265        LaneCount<LANES>: SupportedLaneCount,
266        E: kani_intrinsic::MaskElement,
267        [u8; kani_intrinsic::mask_len(LANES)]: Sized,
268        u64: From<M>,
269    {
270        assert_eq!(
271            unsafe { u64::from(kani_intrinsic::simd_bitmask::<_, M, E, LANES>(mask)) },
272            mask.to_bitmask()
273        );
274    }

276    /// Compare the value returned by our model and the simd_bitmask intrinsic.
277    fn check_bitmask<T, U, E, const LANES: usize>(mask: T)
278    where
279        T: Clone + Copy,
280        U: PartialEq + Debug,
281        E: kani_intrinsic::MaskElement,
282        [u8; kani_intrinsic::mask_len(LANES)]: Sized,
283    {
284        assert_eq!(unsafe { kani_intrinsic::simd_bitmask::<_, U, E, LANES>(mask) }, unsafe {
285            simd_bitmask::<T, U>(mask)
286        });
287    }
